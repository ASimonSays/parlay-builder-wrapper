  function basketballPlayerLedger(summary,name){const wanted=norm(name),rows=[];let points=0,rebounds=0,assists=0,threes=0,blocks=0,steals=0;for(const play of basketballPlays(summary)){const text=clean(play?.text),normalized=norm(text),mentions=Boolean(wanted&&normalized.includes(wanted)),primary=Boolean(wanted&&normalized.startsWith(wanted));if(primary&&play?.scoringPlay){points+=Number(play?.scoreValue||0);if(Number(play?.scoreValue)===3)threes++;}if(mentions&&/rebound/i.test(text))rebounds++;if(mentions&&/assist/i.test(text))assists++;if(mentions&&/block/i.test(text))blocks++;if(mentions&&/steal/i.test(text))steals++;rows.push({time:basketballTime(play),points,rebounds,assists,threes,blocks,steals,pr:points+rebounds,pa:points+assists,ra:rebounds+assists,pra:points+rebounds+assists,doubleCount:[points,rebounds,assists,blocks,steals].filter(value=>value>=10).length});}return rows;}
  function basketballThreshold(summary,leg){const target=Number(leg.target),teamMarket=clean(leg.type).startsWith('team_total');for(const play of basketballPlays(summary)){const away=Number(play?.awayScore),home=Number(play?.homeScore);if(!Number.isFinite(away)||!Number.isFinite(home))continue;const value=teamMarket?basketballScore(play,summary,leg.team):away+home;if(value!=null&&value>target)return{reached:true,time:basketballTime(play),value};}return{reached:false,time:'',value:null};}
  function basketballHeaderScores(summary){const competitors=basketballCompetitors(summary),away=Number(competitors.away.score),home=Number(competitors.home.score);return Number.isFinite(away)&&Number.isFinite(home)?{away,home}:null;}
  function basketballFinalValue(summary,leg){const last=basketballPlays(summary).at(-1);if(last){const value=clean(leg.type).startsWith('team_total')?basketballScore(last,summary,leg.team):Number(last.awayScore)+Number(last.homeScore);if(Number.isFinite(value))return value;}const scores=basketballHeaderScores(summary);if(!scores)return null;if(!clean(leg.type).startsWith('team_total'))return scores.away+scores.home;const side=basketballSide(summary,leg.team);return side==='away'?scores.away:side==='home'?scores.home:null;}
  function basketballFinalMatchup(summary,team){const last=basketballPlays(summary).at(-1);if(last){const mine=basketballScore(last,summary,team),opponent=basketballOpponent(last,summary,team);if(mine!=null&&opponent!=null)return{mine,opponent};}const scores=basketballHeaderScores(summary),side=basketballSide(summary,team);if(!scores||!side)return null;return side==='away'?{mine:scores.away,opponent:scores.home}:{mine:scores.home,opponent:scores.away};}
  function firstHalfSnapshot(summary){const plays=basketballPlays(summary),periodOf=play=>Number(play?.period?.number||play?.period||0),firstHalf=plays.filter(play=>periodOf(play)>0&&periodOf(play)<=2),complete=Boolean(firstHalf.length&&(plays.some(play=>periodOf(play)>=3)||Number(summary?.header?.competitions?.[0]?.status?.period||0)>=3||basketballFinal(summary)));if(!complete)return{complete:false,time:'',play:null};const play=firstHalf.at(-1);return{complete:true,time:basketballTime(play),play};}
  function settleBasketballLeg(leg,summary){
    const type=clean(leg.type),target=Number(leg.target),final=basketballFinal(summary),end=basketballFinalTime(summary);
    if(type==='void')return{status:'VOID',settledAt:end,reason:'Void finalized'};
    if(type==='manual')return{status:'PENDING',settledAt:'',reason:'Manual'};
    const key={player_points:'points',player_rebounds:'rebounds',player_assists:'assists',player_threes:'threes',player_blocks:'blocks',player_points_rebounds:'pr',player_pr:'pr',player_points_assists:'pa',player_pa:'pa',player_rebounds_assists:'ra',player_ra:'ra',player_points_rebounds_assists:'pra',player_pra:'pra',player_double_double:'doubleCount',player_triple_double:'doubleCount'}[type];
    if(key){const needed=type==='player_double_double'?2:type==='player_triple_double'?3:target,reached=threshold(basketballPlayerLedger(summary,leg.player),key,needed);if(reached.reached)return{status:'WIN',settledAt:reached.time,reason:`${key} target reached`};if(final){const value=basketballFinalPlayerValue(summary,leg,key);if(value==null||!Number.isFinite(value))return{status:'UNAVAILABLE',settledAt:'',reason:'Final player statistic unavailable'};return{status:value>=needed?'WIN':'LOSS',settledAt:end,reason:value>=needed?'Target confirmed by final box score':'Game final below target'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(['team_total_over','total_over'].includes(type)){const reached=basketballThreshold(summary,leg);if(reached.reached)return{status:'WIN',settledAt:reached.time,reason:'Over threshold crossed'};if(final){const value=basketballFinalValue(summary,leg);return value==null?{status:'UNAVAILABLE',settledAt:'',reason:'Final score unavailable'}:{status:finalLineStatus(value,target,'over'),settledAt:end,reason:'Game final'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(['team_total_under','total_under'].includes(type)){const loss=basketballThreshold(summary,leg);if(loss.reached)return{status:'LOSS',settledAt:loss.time,reason:'Under threshold crossed'};if(final){const value=basketballFinalValue(summary,leg);return value==null?{status:'UNAVAILABLE',settledAt:'',reason:'Final score unavailable'}:{status:finalLineStatus(value,target,'under'),settledAt:end,reason:'Game final'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(['h1_ml','h1_spread','h1_total_over','h1_total_under'].includes(type)){const half=firstHalfSnapshot(summary);if(!half.complete)return{status:'LIVE',settledAt:'',reason:''};const away=Number(half.play?.awayScore),home=Number(half.play?.homeScore);if(!Number.isFinite(away)||!Number.isFinite(home))return{status:'UNAVAILABLE',settledAt:'',reason:'Halftime score unavailable'};if(type.startsWith('h1_total')){const direction=type.endsWith('_over')?'over':'under';return{status:finalLineStatus(away+home,target,direction),settledAt:half.time,reason:'First half complete'};}const mine=basketballScore(half.play,summary,leg.team),opponent=basketballOpponent(half.play,summary,leg.team);if(mine==null||opponent==null)return{status:'UNAVAILABLE',settledAt:'',reason:'Team unavailable'};const adjusted=type==='h1_spread'?mine+target:mine;return{status:adjusted>opponent?'WIN':adjusted<opponent?'LOSS':'VOID',settledAt:half.time,reason:'First half complete'};}
    if(['ml','spread'].includes(type)){if(!final)return{status:'LIVE',settledAt:'',reason:''};const scores=basketballFinalMatchup(summary,leg.team);if(!scores)return{status:'UNAVAILABLE',settledAt:'',reason:'Final score unavailable'};const adjusted=type==='spread'?scores.mine+target:scores.mine;return{status:adjusted>scores.opponent?'WIN':adjusted<scores.opponent?'LOSS':'VOID',settledAt:end,reason:'Game final'};}
    return{status:final?'UNAVAILABLE':'LIVE',settledAt:'',reason:'Unsupported basketball settlement mapping'};
  }

  function chooseTicketSettlement(results,outcome){const settled=results.filter(result=>result.settledAt);if(outcome==='LOST'){const losses=settled.filter(result=>result.status==='LOSS');return losses.reduce((best,item)=>!best||new Date(item.settledAt)<new Date(best.settledAt)?item:best,null);}if(outcome==='WON'){const wins=settled.filter(result=>result.status==='WIN');return wins.reduce((best,item)=>!best||new Date(item.settledAt)>new Date(best.settledAt)?item:best,null);}if(outcome==='PUSH'){const pushes=settled.filter(result=>result.status==='VOID');return pushes.reduce((best,item)=>!best||new Date(item.settledAt)>new Date(best.settledAt)?item:best,null);}return null;}
  async function resolve(record,outcome){const ticket=record.ticket||{},results=[];for(let index=0;index<(ticket.legs||[]).length;index++){const leg=ticket.legs[index],league=effectiveLeague(ticket,leg),game=leg.game||ticket.game,date=leg.date||ticket.date;if(!supportedLeague(league)||!game||!date){results.push({index,status:'UNAVAILABLE',settledAt:'',reason:'No supported event feed'});continue;}try{const result=league==='MLB'?settleMlbLeg(leg,await getMlbFeed(record,ticket,leg)):settleBasketballLeg(leg,await getBasketballSummary(record,ticket,leg,league));results.push({index,ledger:league==='MLB'?'mlb':'basketball',...result});}catch(error){results.push({index,status:'UNAVAILABLE',settledAt:'',reason:error?.message||'Feed unavailable'});}}return{results,ticketSettlement:chooseTicketSettlement(results,clean(outcome).toUpperCase())};}
  function compact(result){return{index:result.index,status:result.status,settledAt:result.settledAt||null,settlementReason:result.reason||''};}
  async function apply(record,outcome){
    const normalizedOutcome=clean(outcome).toUpperCase(),resolved=await resolve(record,normalizedOutcome),previous=new Map((record.legSettlements||[]).map(item=>[Number(item.index),item]));
    record.legSettlements=resolved.results.map(result=>{const prior=previous.get(Number(result.index));if(result.status==='UNAVAILABLE'&&prior&&['WIN','LOSS','VOID'].includes(prior.status))return prior;const next=compact(result);if(prior&&prior.status===next.status&&!next.settledAt&&prior.settledAt)next.settledAt=prior.settledAt;return next;});
    const settlement=resolved.ticketSettlement;
    if(settlement){record.settledAt=settlement.settledAt;record.settlementSource=settlement.ledger==='basketball'?'basketball-event-ledger':'mlb-event-ledger';record.settlementReason=settlement.reason||'';record.settlementLegIndexes=resolved.results.filter(result=>result.status===settlement.status&&result.settledAt===settlement.settledAt).map(result=>result.index);record.settledOutcome=normalizedOutcome;record.settlementPending=false;}
    else if(record.settledOutcome!==normalizedOutcome){delete record.settledAt;delete record.settlementSource;delete record.settlementReason;delete record.settlementLegIndexes;delete record.settledOutcome;record.settlementPending=['WON','LOST','PUSH'].includes(normalizedOutcome);}
    return record;
  }
  function reset(){cache.clear();}
  window.ParlaySettlementService=Object.freeze({apply,resolve,reset,supportedLeague,supports});
